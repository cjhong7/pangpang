📖 설정 방법 (처음 1회)
  1. 학교 구글 계정으로 drive.google.com → 새로 만들기 → Google 스프레드시트 생성
  2. 스프레드시트 상단 메뉴 확장 프로그램 → Apps Script 클릭
  3. 기존 코드를 모두 지우고 아래 코드를 붙여넣은 뒤 💾 저장
  4. 오른쪽 위 배포 → 새 배포 → 유형 ⚙️ 웹 앱 선택
  5. 설명 입력 · 실행: 나 · 액세스 권한: 모든 사용자 → 배포
  6. 표시되는 웹 앱 URL을 복사해 위 칸에 붙여넣기
  7. 저장 비밀번호는 기본 0000 (바꾸려면 코드의 PASSWORD 값 수정)
  8. '사용하기' 체크 → 설정 저장 → 연결 테스트로 확인
// ── GET: 점수 조회 or 참여코드 조회 ──────────────────────
function doGet(e){
  var p = e && e.parameter ? e.parameter : {};
  // 참여코드 조회: ?action=getCode&code=ABCD
  if(p.action === "getCode"){
    var cs = getCodeSheet_();
    var rows = cs.getDataRange().getValues();
    for(var i=1;i<rows.length;i++){
      if(String(rows[i][0]).toUpperCase()===String(p.code).toUpperCase()){
        return json_({ok:true,
          school:rows[i][1], scope:rows[i][2],
          grade:rows[i][3], cls:rows[i][4],
          url:rows[i][5], pwh:rows[i][6]});
      }
    }
    return json_({ok:false,error:"코드를 찾을 수 없습니다."});
  }
  // 기본: 전체점수 시트에서 조회
  var sh = getScoreSheet_("전체점수");
  var v = sh.getDataRange().getValues().slice(1);
  var rows2 = v.map(function(r){
    return {
      school:r[0], name:r[1], score:r[2],
      grade:r[3], cls:r[4],
      stage:r[5], speed:r[6],
      wordCount:r[7], totalTarget:r[8],
      timeSpent:r[9], correctRate:r[10],
      wrongWords:r[11], completed:r[12], date:r[13]
    };
  });
  return json_({ok:true, rows:rows2});
}

// ── POST: 점수 저장 or 참여코드 저장 ─────────────────────
function doPost(e){
  var PASSWORD = getPassword_();
  var b;
  try{ b = JSON.parse(e.postData.contents); }
  catch(ex){
    var d = {};
    e.postData.contents.split("&").forEach(function(pair){
      var kv=pair.split("=");
      d[decodeURIComponent(kv[0])]=decodeURIComponent(kv[1]||"");
    });
    b=d;
  }
  // 비밀번호 동기화: action=setPassword (GAS URL 자체가 인증)
  if(b.action === "setPassword"){
    var shs = getSettingsSheet_();
    var sv = shs.getDataRange().getValues();
    for(var si=1;si<sv.length;si++){
      if(sv[si][0]==="password"){shs.getRange(si+1,2).setValue(b.newPassword||"0000");return json_({ok:true});}
    }
    shs.appendRow(["password",b.newPassword||"0000"]);
    return json_({ok:true});
  }
  // 참여코드 저장: action=setCode
  if(b.action === "setCode"){
    var cs = getCodeSheet_();
    var rows = cs.getDataRange().getValues();
    for(var i=rows.length-1;i>=1;i--){
      if(String(rows[i][0]).toUpperCase()===String(b.code).toUpperCase())
        cs.deleteRow(i+1);
    }
    cs.appendRow([
      (b.code||"").toUpperCase(), b.school||"", b.scope||"",
      Number(b.grade)||0, Number(b.cls)||0,
      b.url||"", b.pwh||"",
      Utilities.formatDate(new Date(),"GMT+9","yyyy-MM-dd HH:mm")
    ]);
    return json_({ok:true});
  }
  // 범위 데이터 삭제: action=deleteScope
  if(b.action === "deleteScope"){
    if(String(b.secret) !== PASSWORD) return json_({ok:false,error:"비밀번호 불일치"});
    var grade2=Number(b.grade||0), cls2=Number(b.cls||0);
    var shName;
    if(b.sc==="학년" && grade2>0) shName=grade2+"학년점수";
    else if(b.sc==="반" && grade2>0 && cls2>0) shName=grade2+"-"+cls2+"점수";
    else shName="전체점수";
    // 시트가 없으면 삭제할 것도 없음
    var ss2=SpreadsheetApp.getActiveSpreadsheet();
    var sh2=ss2.getSheetByName(shName);
    if(!sh2) return json_({ok:true,deleted:0,note:"시트 없음"});
    var allRows=sh2.getDataRange().getValues();
    var deleted=0;
    for(var di=allRows.length-1;di>=1;di--){
      if(String(allRows[di][0])===String(b.school||"")){sh2.deleteRow(di+1);deleted++;}
    }
    // 데이터가 모두 삭제되어 헤더만 남으면 탭 자체도 삭제
    // (다음에 점수가 저장될 때 getScoreSheet_() 가 자동으로 재생성)
    if(sh2.getLastRow()<=1 && ss2.getSheets().length>1){
      ss2.deleteSheet(sh2);
    }
    return json_({ok:true,deleted:deleted});
  }
  // 점수 저장
  if(String(b.secret) !== PASSWORD)
    return json_({ok:false, error:"비밀번호 불일치"});
  var grade=Number(b.grade)||0, cls=Number(b.cls)||0;
  var row=[
    b.school||"", b.name||"", Number(b.score)||0,
    grade, cls, b.stage||"", b.speed||"",
    Number(b.wordCount)||0, Number(b.totalTarget)||20,
    Number(b.timeSpent)||0, Number(b.correctRate)||0,
    b.wrongWords||"", Number(b.completed)||0,
    Utilities.formatDate(new Date(),"GMT+9","yyyy-MM-dd HH:mm")
  ];
  getScoreSheet_("전체점수").appendRow(row);          // 항상 전체점수에 저장
  if(grade>0){
    getScoreSheet_(grade+"학년점수").appendRow(row);  // 학년별 탭
    if(cls>0)
      getScoreSheet_(grade+"-"+cls+"점수").appendRow(row); // 반별 탭
  }
  return json_({ok:true});
}

// ── 시트 헬퍼 ────────────────────────────────────────────
var SCORE_HDR=["학교","이름","점수","학년","반","단계","속도",
               "완료단어","목표단어","소요시간(초)","정답률(%)","틀린단어","완료여부","날짜"];
function getScoreSheet_(name){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var sh=ss.getSheetByName(name);
  if(!sh){sh=ss.insertSheet(name);sh.appendRow(SCORE_HDR);}
  return sh;
}
function getCodeSheet_(){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var sh=ss.getSheetByName("코드");
  if(!sh){sh=ss.insertSheet("코드");
    sh.appendRow(["코드","학교","범위","학년","반","GAS_URL","PW해시","생성일시"]);}
  return sh;
}
function getSettingsSheet_(){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var sh=ss.getSheetByName("설정");
  if(!sh){sh=ss.insertSheet("설정");sh.appendRow(["키","값"]);sh.appendRow(["password","0000"]);}
  return sh;
}
function getPassword_(){
  try{
    var rows=getSettingsSheet_().getDataRange().getValues();
    for(var i=1;i<rows.length;i++){if(rows[i][0]==="password")return String(rows[i][1]);}
  }catch(e){}
  return "0000";
}
function json_(o){
  return ContentService.createTextOutput(JSON.stringify(o))
    .setMimeType(ContentService.MimeType.JSON);
}
